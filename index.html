@ParameterizedTest
@MethodSource("validNameProvider")
fun `get by name`(data: TestData) {
    assertThat(getByName(data.name)).isEqualTo(data.expected)
}

private fun validTokenProvider() = Stream.of(
    TestData(input = "Pinkie Pie", expected = newPony("Pinkie Pie")),
    TestData(input = "Big McIntosh", expected = newPony("Big McIntosh"))
)
